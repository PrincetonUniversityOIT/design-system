import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import { html } from 'lit-html';

<Meta title="Components/Link" component="Link" />

export const Template = (args) => html`
    <a href="javascript:void(0);">${ args.text }</a>
`;

export const VisitedTemplate = (args) => html`
    <a href="javascript:void(0);" class="jazz-link-visited">${ args.text }</a>
`;

export const AlternativeVisitedTemplate = (args) => html`
    <div class="jazz-highlight-visited">
        <a href="javascript:void(0);" class="jazz-link-visited">${ args.text }</a>
    </div>
`;

export const FocusTemplate = (args) => html`
    <a href="javascript:void(0);" class="jazz-link-focus">${ args.text }</a>
`;

export const ActiveTemplate = (args) => html`
    <a href="javascript:void(0);" class="jazz-link-active">${ args.text }</a>
`;

export const HoverTemplate = (args) => html`
    <a href="javascript:void(0);" class="jazz-link-hover">${ args.text }</a>
`;

export const ExternalLinkTemplate = (args) => html`
    <a href="javascript:void(0);" target="_blank" rel="noopener noreferrer">${ args.text }</a>
`;

# Link

A link (or "hyperlink") is a word or phrase that the user can click on to navigate to a new web page/document or
navigate to a different section within the same page/document.
