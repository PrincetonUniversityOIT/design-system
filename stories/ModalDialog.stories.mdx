export const PassiveDialogTemplate = (args) => html`
    <script id="initscript">
        var modalPassive = document.getElementById('passiveTest');
        window.onclick = function(event) {
            if (event.target == modalPassive) {
                modalPassive.classList.remove("emc-modal__wrapper--visible");
            }
        }
        function toggleDialogPassive() {
            modalPassive.classList.toggle("emc-modal__wrapper--visible");
        }
    </script>
    <button class="emc-btn" onClick="toggleDialogPassive()">Test Passive Dialog!</button>
    <div id="passiveTest" role="presentation" class="emc-modal-wrapper">
      <div role="dialog" aria-label="Passive Dialog" aria-modal="true" class="emc-modal emc-modal--sm">
      <div class="emc-modal-title">
                ${ args.title }
                  <button class="emc-modal-button__close emc-modal-button--transparent"
                      aria-label="Close modal dialog" onClick="toggleDialogPassive()"></button>
        </div>
        <div class="emc-modal-content" aria-labelledBy="emc-modal-title">${ args.content }</div>
      </div>
    </div>
`;

export const TransactionalModalComponentDialogTemplate = (args) => html`
    <script id="initscript">
        var modalTransaction = document.getElementById("transactionModal");
        modalTransaction.addEventListener("ok", function () {
            console.log("ok");
        });
        modalTransaction.addEventListener("cancel", function () {
            console.log("cancel");
        });
         var btn = document.getElementById("btnTest");
         btn.addEventListener("click", function () {
             modalTransaction.visible = !modal.visible;
         });
    </script>
    <button class="emc-btn" id="btnTest">Test Me!</button>
    <emc-modal-dialog id="transactionModal" visible="false" title="${ args.title}">
        ${ args.content }
    </emc-modal-dialog>
`;
